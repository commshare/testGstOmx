50 /*
51  * This routine determines what format the input file is,
52  * and selects the appropriate input-reading module.
53  *
54  * To determine which family of input formats the file belongs to,
55  * we may look only at the first byte of the file, since C does not
56  * guarantee that more than one character can be pushed back with ungetc.
57  * Looking at additional bytes would require one of these approaches:
58  * 1) assume we can fseek() the input file (fails for piped input);
59  * 2) assume we can push back more than one character (works in
60  * some C implementations, but unportable);
61  * 3) provide our own buffering (breaks input readers that want to use
62  * stdio directly, such as the RLE library);
63  * or 4) don't put back the data, and modify the input_init methods to assume
64  * they start reading after the start of file (also breaks RLE library).
65  * #1 is attractive for MS-DOS but is untenable on Unix.
66  *
67  * The most portable solution for file types that can't be identified by their
68  * first byte is to make the user tell us what they are. This is also the
69  * only approach for "raw" file types that contain only arbitrary values.
70  * We presently apply this method for Targa files. Most of the time Targa
71  * files start with 0x00, so we recognize that case. Potentially, however,
72  * a Targa file could start with any byte value (byte 0 is the length of the
73  * seldom-used ID field), so we provide a switch to force Targa input mode.
74  */

202 parse_switches (j_compress_ptr cinfo, int argc, char **argv,
203  int last_file_arg_seen, boolean for_real)
204 /* Parse optional switches.
205  * Returns argv[] index of first file-name argument (== argc if none).
206  * Any file names with indexes <= last_file_arg_seen are ignored;
207  * they have presumably been processed in a previous iteration.
208  * (Pass 0 for last_file_arg_seen on the first or only iteration.)
209  * for_real is FALSE on the first (dummy) pass; we may skip any expensive
210  * processing.
211  */
212 {
213  int argn;
214  char * arg;
215  boolean force_baseline;
216  boolean simple_progressive;
217  char * qualityarg = NULL; /* saves -quality parm if any */
218  char * qtablefile = NULL; /* saves -qtables filename if any */
219  char * qslotsarg = NULL; /* saves -qslots parm if any */
220  char * samplearg = NULL; /* saves -sample parm if any */
221  char * scansarg = NULL; /* saves -scans parm if any */
222 
223  /* Set up default JPEG parameters. */
224 
225  force_baseline = FALSE; /* by default, allow 16-bit quantizers */
226  simple_progressive = FALSE;
227  is_targa = FALSE;
228  outfilename = NULL;
229  cinfo->err->trace_level = 0;
230 
231  /* Scan command line options, adjust parameters */
232 
233  for (argn = 1; argn < argc; argn++) {
234  arg = argv[argn];
235  if (*arg != '-') {
236  /* Not a switch, must be a file name argument */
237  if (argn <= last_file_arg_seen) {
238  outfilename = NULL; /* -outfile applies to just one input file */
239  continue; /* ignore this name if previously processed */
240  }
241  break; /* else done parsing switches */
242  }
243  arg++; /* advance past switch marker character */
244 
245  if (keymatch(arg, "arithmetic", 1)) {
246  /* Use arithmetic coding. */
247 #ifdef C_ARITH_CODING_SUPPORTED
248  cinfo->arith_code = TRUE;
249 #else
250  fprintf(stderr, "%s: sorry, arithmetic coding not supported\n",
251  progname);
252  exit(EXIT_FAILURE);
253 #endif
254 
255  } else if (keymatch(arg, "baseline", 1)) {
256  /* Force baseline-compatible output (8-bit quantizer values). */
257  force_baseline = TRUE;
258 
259  } else if (keymatch(arg, "dct", 2)) {
260  /* Select DCT algorithm. */
261  if (++argn >= argc) /* advance to next argument */
262  usage();
263  if (keymatch(argv[argn], "int", 1)) {
264  cinfo->dct_method = JDCT_ISLOW;
265  } else if (keymatch(argv[argn], "fast", 2)) {
266  cinfo->dct_method = JDCT_IFAST;
267  } else if (keymatch(argv[argn], "float", 2)) {
268  cinfo->dct_method = JDCT_FLOAT;
269  } else
270  usage();
271 
272  } else if (keymatch(arg, "debug", 1) || keymatch(arg, "verbose", 1)) {
273  /* Enable debug printouts. */
274  /* On first -d, print version identification */
275  static boolean printed_version = FALSE;
276 
277  if (! printed_version) {
278  fprintf(stderr, "Independent JPEG Group's CJPEG, version %s\n%s\n",
279  JVERSION, JCOPYRIGHT);
280  printed_version = TRUE;
281  }
282  cinfo->err->trace_level++;
283 
284  } else if (keymatch(arg, "grayscale", 2) || keymatch(arg, "greyscale",2)) {
285  /* Force a monochrome JPEG file to be generated. */
286  jpeg_set_colorspace(cinfo, JCS_GRAYSCALE);
287 
288  } else if (keymatch(arg, "maxmemory", 3)) {
289  /* Maximum memory in Kb (or Mb with 'm'). */
290  long lval;
291  char ch = 'x';
292 
293  if (++argn >= argc) /* advance to next argument */
294  usage();
295  if (sscanf(argv[argn], "%ld%c", &lval, &ch) < 1)
296  usage();
297  if (ch == 'm' || ch == 'M')
298  lval *= 1000L;
299  cinfo->mem->max_memory_to_use = lval * 1000L;
300 
301  } else if (keymatch(arg, "optimize", 1) || keymatch(arg, "optimise", 1)) {
302  /* Enable entropy parm optimization. */
303 #ifdef ENTROPY_OPT_SUPPORTED
304  cinfo->optimize_coding = TRUE;
305 #else
306  fprintf(stderr, "%s: sorry, entropy optimization was not compiled\n",
307  progname);
308  exit(EXIT_FAILURE);
309 #endif
310 
311  } else if (keymatch(arg, "outfile", 4)) {
312  /* Set output file name. */
313  if (++argn >= argc) /* advance to next argument */
314  usage();
315  outfilename = argv[argn]; /* save it away for later use */
316 
317  } else if (keymatch(arg, "progressive", 1)) {
318  /* Select simple progressive mode. */
319 #ifdef C_PROGRESSIVE_SUPPORTED
320  simple_progressive = TRUE;
321  /* We must postpone execution until num_components is known. */
322 #else
323  fprintf(stderr, "%s: sorry, progressive output was not compiled\n",

327 
328  } else if (keymatch(arg, "quality", 1)) {

333 
334  } else if (keymatch(arg, "qslots", 2)) {

349  /* We postpone actually reading the file in case -quality comes later. */

351  } else if (keymatch(arg, "restart", 1)) {

354  char ch = 'x';
355 
356  if (++argn >= argc) /* advance to next argument */
357  usage();

365  } else {

369 
370  } else if (keymatch(arg, "sample", 2)) {
371  /* Set sampling factors. */
372  if (++argn >= argc) /* advance to next argument */
373  usage();
374  samplearg = argv[argn];
375  /* Must delay setting sample factors until after we have processed any
376  * colorspace-determining switches, since jpeg_set_colorspace sets
377  * default sampling factors.
378  */
379 
380  } else if (keymatch(arg, "scale", 4)) {
381  /* Scale the image by a fraction M/N. */
382  if (++argn >= argc) /* advance to next argument */
383  usage();
384  if (sscanf(argv[argn], "%d/%d",
385  &cinfo->scale_num, &cinfo->scale_denom) != 2)
386  usage();
387 
388  } else if (keymatch(arg, "scans", 4)) {
389  /* Set scan script. */
390 #ifdef C_MULTISCAN_FILES_SUPPORTED
391  if (++argn >= argc) /* advance to next argument */
392  usage();
393  scansarg = argv[argn];
394  /* We must postpone reading the file in case -progressive appears. */
395 #else
396  fprintf(stderr, "%s: sorry, multi-scan output was not compiled\n",
397  progname);
398  exit(EXIT_FAILURE);
399 #endif
400 
401  } else if (keymatch(arg, "smooth", 2)) {
402  /* Set input smoothing factor. */
403  int val;
404 
405  if (++argn >= argc) /* advance to next argument */
406  usage();
407  if (sscanf(argv[argn], "%d", &val) != 1)
408  usage();
409  if (val < 0 || val > 100)
410  usage();
411  cinfo->smoothing_factor = val;
412 
413  } else if (keymatch(arg, "targa", 1)) {
414  /* Input file is Targa format. */
415  is_targa = TRUE;
416 
417  } else {
418  usage(); /* bogus switch */
419  }
420  }
421 
422  /* Post-switch-scanning cleanup */
423 
424  if (for_real) {
425 
426  /* Set quantization tables for selected quality. */
427  /* Some or all may be overridden if -qtables is present. */
428  if (qualityarg != NULL) /* process -quality if it was present */
429  if (! set_quality_ratings(cinfo, qualityarg, force_baseline))
430  usage();
431 
432  if (qtablefile != NULL) /* process -qtables if it was present */
433  if (! read_quant_tables(cinfo, qtablefile, force_baseline))
434  usage();
435 
436  if (qslotsarg != NULL) /* process -qslots if it was present */
437  if (! set_quant_slots(cinfo, qslotsarg))
438  usage();
439 
440  if (samplearg != NULL) /* process -sample if it was present */
441  if (! set_sample_factors(cinfo, samplearg))
442  usage();
443 
444 #ifdef C_PROGRESSIVE_SUPPORTED
445  if (simple_progressive) /* process -progressive; -scans can override */
446  jpeg_simple_progression(cinfo);
447 #endif
448 
449 #ifdef C_MULTISCAN_FILES_SUPPORTED
450  if (scansarg != NULL) /* process -scans if it was present */
451  if (! read_scan_script(cinfo, scansarg))
452  usage();
453 #endif
454  }
455 
456  return argn; /* return index of next arg (file name) */
457 }

466 {

468  struct jpeg_error_mgr jerr;

481 #endif
482 
483  progname = argv[0];
484  if (progname == NULL || progname[0] == 0)
485  progname = "cjpeg"; /* in case C library doesn't provide it */
486 
487  /* Initialize the JPEG compression object with default error handling. */
488  cinfo.err = jpeg_std_error(&jerr);
489  jpeg_create_compress(&cinfo);
490  /* Add some application-specific error messages (from cderror.h) */
491  jerr.addon_message_table = cdjpeg_message_table;
492  jerr.first_addon_message = JMSG_FIRSTADDONCODE;
493  jerr.last_addon_message = JMSG_LASTADDONCODE;
494 
495  /* Now safe to enable signal catcher. */
496 #ifdef NEED_SIGNAL_CATCHER
497  enable_signal_catcher((j_common_ptr) &cinfo);
498 #endif
499 
500  /* Initialize JPEG parameters.
501  * Much of this may be overridden later.
502  * In particular, we don't yet know the input file's color space,
503  * but we need to provide some value for jpeg_set_defaults() to work.
504  */
505 
506  cinfo.in_color_space = JCS_RGB; /* arbitrary guess */
507  jpeg_set_defaults(&cinfo);
508 
509  /* Scan command line to find file names.
510  * It is convenient to use just one switch-parsing routine, but the switch
511  * values read here are ignored; we will rescan the switches after opening
512  * the input file.
513  */
514 
515  file_index = parse_switches(&cinfo, argc, argv, 0, FALSE);
516 
517 #ifdef TWO_FILE_COMMANDLINE
518  /* Must have either -outfile switch or explicit output file name */
519  if (outfilename == NULL) {
520  if (file_index != argc-2) {
521  fprintf(stderr, "%s: must name one input and one output file\n",
522  progname);
523  usage();
524  }
525  outfilename = argv[file_index+1];
526  } else {
527  if (file_index != argc-1) {
528  fprintf(stderr, "%s: must name one input and one output file\n",
529  progname);
530  usage();
531  }
532  }
533 #else
534  /* Unix style: expect zero or one file name */
535  if (file_index < argc-1) {
536  fprintf(stderr, "%s: only one input file\n", progname);
537  usage();
538  }
539 #endif /* TWO_FILE_COMMANDLINE */
540 
541  /* Open the input file. */
542  if (file_index < argc) {
543  if ((input_file = fopen(argv[file_index], READ_BINARY)) == NULL) {
544  fprintf(stderr, "%s: can't open %s\n", progname, argv[file_index]);
545  exit(EXIT_FAILURE);
546  }
547  } else {
548  /* default input file is stdin */
549  input_file = read_stdin();
550  }
551 
552  /* Open the output file. */
553  if (outfilename != NULL) {
554  if ((output_file = fopen(outfilename, WRITE_BINARY)) == NULL) {
555  fprintf(stderr, "%s: can't open %s\n", progname, outfilename);
556  exit(EXIT_FAILURE);
557  }
558  } else {
559  /* default output file is stdout */
560  output_file = write_stdout();
561  }
562 
563 #ifdef PROGRESS_REPORT
564  start_progress_monitor((j_common_ptr) &cinfo, &progress);
565 #endif
566 
567  /* Figure out the input file format, and set up to read it. */
568  src_mgr = select_file_type(&cinfo, input_file);
569  src_mgr->input_file = input_file;
570 
571  /* Read the input file header to obtain file size & colorspace. */
572  (*src_mgr->start_input) (&cinfo, src_mgr);
573 
574  /* Now that we know input colorspace, fix colorspace-dependent defaults */
575  jpeg_default_colorspace(&cinfo);
576 
577  /* Adjust default compression parameters by re-parsing the options */
578  file_index = parse_switches(&cinfo, argc, argv, 0, TRUE);
579 
580  /* Specify data destination for compression */
581  jpeg_stdio_dest(&cinfo, output_file);
582 
583  /* Start compressor */
584  jpeg_start_compress(&cinfo, TRUE);
585 
586  /* Process data */
587  while (cinfo.next_scanline < cinfo.image_height) {
588  num_scanlines = (*src_mgr->get_pixel_rows) (&cinfo, src_mgr);
589  (void) jpeg_write_scanlines(&cinfo, src_mgr->buffer, num_scanlines);
590  }
591 
592  /* Finish compression and release memory */
593  (*src_mgr->finish_input) (&cinfo, src_mgr);
594  jpeg_finish_compress(&cinfo);
595  jpeg_destroy_compress(&cinfo);
596 
597  /* Close files, if we opened them */
598  if (input_file != stdin)
599  fclose(input_file);
600  if (output_file != stdout)
601  fclose(output_file);
602 
603 #ifdef PROGRESS_REPORT
604  end_progress_monitor((j_common_ptr) &cinfo);
605 #endif
606 
607  /* All done. */
608  exit(jerr.num_warnings ? EXIT_WARNING : EXIT_SUCCESS);
609  return 0; /* suppress no-return-value warnings */
610 }
