166  OMX_S32 denominator=0;
167  OMX_U32 i,sampleCount = pInBuffer->nFilledLen / 2; // signed 16 bit samples assumed
168  omx_audio_mixer_component_PrivateType* omx_audio_mixer_component_Private = openmaxStandComp->pComponentPrivate;
169  omx_audio_mixer_component_PortType* pPort;
170 
171  for(i=0;i<omx_audio_mixer_component_Private->sPortTypesParam[OMX_PortDomainAudio].nPorts-1;i++) {
172  pPort = (omx_audio_mixer_component_PortType*)omx_audio_mixer_component_Private->ports[i];
173  if(PORT_IS_ENABLED(pPort)){
174  denominator+=pPort->sVolume.sVolume.nValue;
175  }
176  }
177 
178  pPort = (omx_audio_mixer_component_PortType*)omx_audio_mixer_component_Private->ports[pInBuffer->nInputPortIndex];
179 
180  /*Copy the first buffer with appropriate gain*/
181  if(pOutBuffer->nFilledLen == 0) {
182  memset(pOutBuffer->pBuffer,0,pInBuffer->nFilledLen);
183 
184  for (i = 0; i < sampleCount; i++) {
185  ((OMX_S16*) pOutBuffer->pBuffer)[i] = (OMX_S16)
186  ((((OMX_S16*) pInBuffer->pBuffer)[i] * pPort->sVolume.sVolume.nValue ) / denominator);
187  }
188  } else { // For the second buffer add with the first buffer with gain
189  for (i = 0; i < sampleCount; i++) {
190  ((OMX_S16*) pOutBuffer->pBuffer)[i] += (OMX_S16)
191  ((((OMX_S16*) pInBuffer->pBuffer)[i] * pPort->sVolume.sVolume.nValue ) / denominator);
192  }
193  }
194 
195  pOutBuffer->nFilledLen = pInBuffer->nFilledLen;
196  pInBuffer->nFilledLen=0;
