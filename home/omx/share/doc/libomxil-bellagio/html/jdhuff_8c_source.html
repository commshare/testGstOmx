50 /*
51  * Fetching the next N bits from the input stream is a time-critical operation
52  * for the Huffman decoders. We implement it with a combination of inline
53  * macros and out-of-line subroutines. Note that N (the number of bits
54  * demanded at one time) never exceeds 15 for JPEG use.
55  *
56  * We read source bytes into get_buffer and dole out bits as needed.
57  * If get_buffer already contains enough bits, they are fetched in-line
58  * by the macros CHECK_BIT_BUFFER and GET_BITS. When there aren't enough
59  * bits, jpeg_fill_bit_buffer is called; it will attempt to fill get_buffer
60  * as full as possible (not just to the number of bits needed; this
61  * prefetching reduces the overhead cost of calling jpeg_fill_bit_buffer).
62  * Note that jpeg_fill_bit_buffer may return FALSE to indicate suspension.
63  * On TRUE return, jpeg_fill_bit_buffer guarantees that get_buffer contains
64  * at least the requested number of bits --- dummy zeroes are inserted if
65  * necessary.
66  */

116 /*
117  * These macros provide the in-line portion of bit fetching.
118  * Use CHECK_BIT_BUFFER to ensure there are N bits in get_buffer
119  * before using GET_BITS, PEEK_BITS, or DROP_BITS.
120  * The variables get_buffer and bits_left are assumed to be locals,
121  * but the state struct might not be (jpeg_huff_decode needs this).
122  * CHECK_BIT_BUFFER(state,n,action);
123  * Ensure there are N bits in get_buffer; if suspend, take action.
124  * val = GET_BITS(n);
125  * Fetch next N bits.
126  * val = PEEK_BITS(n);
127  * Fetch next N bits without removing them from the buffer.
128  * DROP_BITS(n);
129  * Discard next N bits.
130  * The value N should be a simple variable, not an expression, because it
131  * is evaluated multiple times.
132  */
133 
134 #define CHECK_BIT_BUFFER(state,nbits,action) \
135  { if (bits_left < (nbits)) { \
136  if (! jpeg_fill_bit_buffer(&(state),get_buffer,bits_left,nbits)) \
137  { action; } \
138  get_buffer = (state).get_buffer; bits_left = (state).bits_left; } }
139 
140 #define GET_BITS(nbits) \
141  (((int) (get_buffer >> (bits_left -= (nbits)))) & BIT_MASK(nbits))
142 
143 #define PEEK_BITS(nbits) \
144  (((int) (get_buffer >> (bits_left - (nbits)))) & BIT_MASK(nbits))
145 
146 #define DROP_BITS(nbits) \
147  (bits_left -= (nbits))

150 /*
151  * Code for extracting next Huffman-coded symbol from input bit stream.
152  * Again, this is time-critical and we make the main paths be macros.
153  *
154  * We use a lookahead table to process codes of up to HUFF_LOOKAHEAD bits
155  * without looping. Usually, more than 95% of the Huffman codes will be 8
156  * or fewer bits long. The few overlength codes are handled with a loop,
157  * which need not be inline code.
158  *
159  * Notes about the HUFF_DECODE macro:
160  * 1. Near the end of the data segment, we may fail to get enough bits
161  * for a lookahead. In that case, we do it the hard way.
162  * 2. If the lookahead table contains no entry, the next code must be
163  * more than HUFF_LOOKAHEAD bits long.
164  * 3. jpeg_huff_decode returns -1 if forced to suspend.
165  */
166 
167 #define HUFF_DECODE(result,state,htbl,failaction,slowlabel) \
168 { register int nb, look; \
169  if (bits_left < HUFF_LOOKAHEAD) { \
170  if (! jpeg_fill_bit_buffer(&state,get_buffer,bits_left, 0)) {failaction;} \
171  get_buffer = state.get_buffer; bits_left = state.bits_left; \
172  if (bits_left < HUFF_LOOKAHEAD) { \
173  nb = 1; goto slowlabel; \
174  } \
175  } \
176  look = PEEK_BITS(HUFF_LOOKAHEAD); \
177  if ((nb = htbl->look_nbits[look]) != 0) { \
178  DROP_BITS(nb); \
179  result = htbl->look_sym[look]; \
180  } else { \
181  nb = HUFF_LOOKAHEAD+1; \
182 slowlabel: \
183  if ((result=jpeg_huff_decode(&state,get_buffer,bits_left,htbl,nb)) < 0) \
184  { failaction; } \
185  get_buffer = state.get_buffer; bits_left = state.bits_left; \
186  } \
187 }

275 LOCAL(void)
276 jpeg_make_d_derived_tbl (j_decompress_ptr cinfo, boolean isDC, int tblno,
277  d_derived_tbl ** pdtbl)
278 {
279  JHUFF_TBL *htbl;
280  d_derived_tbl *dtbl;
281  int p, i, l, si, numsymbols;
282  int lookbits, ctr;
283  char huffsize[257];
284  unsigned int huffcode[257];
285  unsigned int code;
286 
287  /* Note that huffsize[] and huffcode[] are filled in code-length order,
288  * paralleling the order of the symbols themselves in htbl->huffval[].
289  */
290 
291  /* Find the input Huffman table */
292  if (tblno < 0 || tblno >= NUM_HUFF_TBLS)
293  ERREXIT1(cinfo, JERR_NO_HUFF_TABLE, tblno);
294  htbl =
295  isDC ? cinfo->dc_huff_tbl_ptrs[tblno] : cinfo->ac_huff_tbl_ptrs[tblno];
296  if (htbl == NULL)
297  ERREXIT1(cinfo, JERR_NO_HUFF_TABLE, tblno);
298 
299  /* Allocate a workspace if we haven't already done so. */
300  if (*pdtbl == NULL)
301  *pdtbl = (d_derived_tbl *)
302  (*cinfo->mem->alloc_small) ((j_common_ptr) cinfo, JPOOL_IMAGE,
303  SIZEOF(d_derived_tbl));
304  dtbl = *pdtbl;
305  dtbl->pub = htbl; /* fill in back link */
306 
307  /* Figure C.1: make table of Huffman code length for each symbol */
308 
309  p = 0;
310  for (l = 1; l <= 16; l++) {
311  i = (int) htbl->bits[l];
312  if (i < 0 || p + i > 256) /* protect against table overrun */
313  ERREXIT(cinfo, JERR_BAD_HUFF_TABLE);
314  while (i--)
315  huffsize[p++] = (char) l;
316  }
317  huffsize[p] = 0;
318  numsymbols = p;
319 
320  /* Figure C.2: generate the codes themselves */
321  /* We also validate that the counts represent a legal Huffman code tree. */
322 
323  code = 0;
324  si = huffsize[0];
325  p = 0;
326  while (huffsize[p]) {
327  while (((int) huffsize[p]) == si) {
328  huffcode[p++] = code;
329  code++;
330  }
331  /* code is now 1 more than the last code used for codelength si; but
332  * it must still fit in si bits, since no code is allowed to be all ones.
333  */
334  if (((INT32) code) >= (((INT32) 1) << si))
335  ERREXIT(cinfo, JERR_BAD_HUFF_TABLE);
336  code <<= 1;
337  si++;
338  }
339 
340  /* Figure F.15: generate decoding tables for bit-sequential decoding */
341 
342  p = 0;
343  for (l = 1; l <= 16; l++) {
344  if (htbl->bits[l]) {
345  /* valoffset[l] = huffval[] index of 1st symbol of code length l,
346  * minus the minimum code of length l
347  */
348  dtbl->valoffset[l] = (INT32) p - (INT32) huffcode[p];
349  p += htbl->bits[l];
350  dtbl->maxcode[l] = huffcode[p-1]; /* maximum code of length l */
351  } else {
352  dtbl->maxcode[l] = -1; /* -1 if no codes of this length */
353  }
354  }
355  dtbl->maxcode[17] = 0xFFFFFL; /* ensures jpeg_huff_decode terminates */
356 
357  /* Compute lookahead tables to speed up decoding.
358  * First we set all the table entries to 0, indicating "too long";
359  * then we iterate through the Huffman codes that are short enough and
360  * fill in all the entries that correspond to bit sequences starting
361  * with that code.
362  */
363 
364  MEMZERO(dtbl->look_nbits, SIZEOF(dtbl->look_nbits));
365 
366  p = 0;
367  for (l = 1; l <= HUFF_LOOKAHEAD; l++) {
368  for (i = 1; i <= (int) htbl->bits[l]; i++, p++) {
369  /* l = current code's length, p = its index in huffcode[] & huffval[]. */
370  /* Generate left-justified code followed by all possible bit sequences */
371  lookbits = huffcode[p] << (HUFF_LOOKAHEAD-l);
372  for (ctr = 1 << (HUFF_LOOKAHEAD-l); ctr > 0; ctr--) {
373  dtbl->look_nbits[lookbits] = l;
374  dtbl->look_sym[lookbits] = htbl->huffval[p];
375  lookbits++;
376  }
377  }
378  }
379 
380  /* Validate symbols as being reasonable.
381  * For AC tables, we make no check, but accept all byte values 0..255.
382  * For DC tables, we require the symbols to be in range 0..15.
383  * (Tighter bounds could be applied depending on the data depth and mode,
384  * but this is sufficient to ensure safe decoding.)
385  */
386  if (isDC) {
387  for (i = 0; i < numsymbols; i++) {
388  int sym = htbl->huffval[i];
389  if (sym < 0 || sym > 15)
390  ERREXIT(cinfo, JERR_BAD_HUFF_TABLE);
391  }
392  }
393 }

399  * but are returned in the corresponding fields of the state struct.

401  * On most machines MIN_GET_BITS should be 25 to allow the full 32-bit width

408  */

417 LOCAL(boolean)
418 jpeg_fill_bit_buffer (bitread_working_state * state,
419  register bit_buf_type get_buffer, register int bits_left,
420  int nbits)
421 /* Load up the bit buffer to a depth of at least nbits */
422 {
423  /* Copy heavily used state fields into locals (hopefully registers) */
424  register const JOCTET * next_input_byte = state->next_input_byte;
425  register size_t bytes_in_buffer = state->bytes_in_buffer;
426  j_decompress_ptr cinfo = state->cinfo;
427 
428  /* Attempt to load at least MIN_GET_BITS bits into get_buffer. */
429  /* (It is assumed that no request will be for more than that many bits.) */
430  /* We fail to do so only if we hit a marker or are forced to suspend. */
431 
432  if (cinfo->unread_marker == 0) { /* cannot advance past a marker */
433  while (bits_left < MIN_GET_BITS) {
434  register int c;
435 
436  /* Attempt to read a byte */
437  if (bytes_in_buffer == 0) {
438  if (! (*cinfo->src->fill_input_buffer) (cinfo))
439  return FALSE;
440  next_input_byte = cinfo->src->next_input_byte;
441  bytes_in_buffer = cinfo->src->bytes_in_buffer;
442  }
443  bytes_in_buffer--;
444  c = GETJOCTET(*next_input_byte++);
445 
446  /* If it's 0xFF, check and discard stuffed zero byte */
447  if (c == 0xFF) {
448  /* Loop here to discard any padding FF's on terminating marker,
449  * so that we can save a valid unread_marker value. NOTE: we will
450  * accept multiple FF's followed by a 0 as meaning a single FF data
451  * byte. This data pattern is not valid according to the standard.
452  */
453  do {
454  if (bytes_in_buffer == 0) {
455  if (! (*cinfo->src->fill_input_buffer) (cinfo))
456  return FALSE;
457  next_input_byte = cinfo->src->next_input_byte;
458  bytes_in_buffer = cinfo->src->bytes_in_buffer;
459  }
460  bytes_in_buffer--;
461  c = GETJOCTET(*next_input_byte++);
462  } while (c == 0xFF);
463 
464  if (c == 0) {
465  /* Found FF/00, which represents an FF data byte */
466  c = 0xFF;
467  } else {
468  /* Oops, it's actually a marker indicating end of compressed data.
469  * Save the marker code for later use.
470  * Fine point: it might appear that we should save the marker into
471  * bitread working state, not straight into permanent state. But
472  * once we have hit a marker, we cannot need to suspend within the
473  * current MCU, because we will read no more bytes from the data
474  * source. So it is OK to update permanent state right away.
475  */
476  cinfo->unread_marker = c;
477  /* See if we need to insert some fake zero bits. */
478  goto no_more_bytes;
479  }
480  }
481 
482  /* OK, load c into get_buffer */
483  get_buffer = (get_buffer << 8) | c;
484  bits_left += 8;
485  } /* end while */
486  } else {
487  no_more_bytes:
488  /* We get here if we've read the marker that terminates the compressed
489  * data segment. There should be enough bits in the buffer register
490  * to satisfy the request; if so, no problem.
491  */
492  if (nbits > bits_left) {
493  /* Uh-oh. Report corrupted data to user and stuff zeroes into
494  * the data stream, so that we can produce some kind of image.
495  * We use a nonvolatile flag to ensure that only one warning message
496  * appears per data segment.
497  */
498  if (! cinfo->entropy->insufficient_data) {
499  WARNMS(cinfo, JWRN_HIT_MARKER);
500  cinfo->entropy->insufficient_data = TRUE;
501  }
502  /* Fill the buffer with zero bits */
503  get_buffer <<= MIN_GET_BITS - bits_left;
504  bits_left = MIN_GET_BITS;
505  }
506  }
507 
508  /* Unload the local registers */
509  state->next_input_byte = next_input_byte;
510  state->bytes_in_buffer = bytes_in_buffer;
511  state->get_buffer = get_buffer;
512  state->bits_left = bits_left;
513 
514  return TRUE;
515 }

544 LOCAL(int)
545 jpeg_huff_decode (bitread_working_state * state,
546  register bit_buf_type get_buffer, register int bits_left,
547  d_derived_tbl * htbl, int min_bits)
548 {
549  register int l = min_bits;
550  register INT32 code;
551 
552  /* HUFF_DECODE has determined that the code is at least min_bits */
553  /* bits long, so fetch that many bits in one swoop. */
554 
555  CHECK_BIT_BUFFER(*state, l, return -1);
556  code = GET_BITS(l);
557 
558  /* Collect the rest of the Huffman code one bit at a time. */
559  /* This is per Figure F.16 in the JPEG spec. */
560 
561  while (code > htbl->maxcode[l]) {
562  code <<= 1;
563  CHECK_BIT_BUFFER(*state, 1, return -1);
564  code |= GET_BITS(1);
565  l++;
566  }
567 
568  /* Unload the local registers */
569  state->get_buffer = get_buffer;
570  state->bits_left = bits_left;
571 
572  /* With garbage input we may reach the sentinel value l = 17. */
573 
574  if (l > 16) {
575  WARNMS(state->cinfo, JWRN_HUFF_BAD_CODE);
576  return 0; /* fake a zero as the safest result */
577  }
578 
579  return htbl->pub->huffval[ (int) (code + htbl->valoffset[l]) ];
580 }

583 /*
584  * Check for a restart marker & resynchronize decoder.
585  * Returns FALSE if must suspend.
586  */
587 
588 LOCAL(boolean)
589 process_restart (j_decompress_ptr cinfo)
590 {
591  huff_entropy_ptr entropy = (huff_entropy_ptr) cinfo->entropy;
592  int ci;
593 
594  /* Throw away any unused bits remaining in bit buffer; */
595  /* include any full bytes in next_marker's count of discarded bytes */
596  cinfo->marker->discarded_bytes += entropy->bitstate.bits_left / 8;
597  entropy->bitstate.bits_left = 0;
598 
599  /* Advance past the RSTn marker */
600  if (! (*cinfo->marker->read_restart_marker) (cinfo))
601  return FALSE;
602 
603  /* Re-initialize DC predictions to 0 */
604  for (ci = 0; ci < cinfo->comps_in_scan; ci++)
605  entropy->saved.last_dc_val[ci] = 0;
606  /* Re-init EOB run count, too */
607  entropy->saved.EOBRUN = 0;
608 
609  /* Reset restart counter */
610  entropy->restarts_to_go = cinfo->restart_interval;
611 
612  /* Reset out-of-data flag, unless read_restart_marker left us smack up
613  * against a marker. In that case we will end up treating the next data
614  * segment as empty, and we can avoid producing bogus output pixels by
615  * leaving the flag set.
616  */
617  if (cinfo->unread_marker == 0)
618  entropy->pub.insufficient_data = FALSE;
619 
620  return TRUE;
621 }

624 /*
625  * Huffman MCU decoding.
626  * Each of these routines decodes and returns one MCU's worth of
627  * Huffman-compressed coefficients.
628  * The coefficients are reordered from zigzag order into natural array order,
629  * but are not dequantized.
630  *
631  * The i'th block of the MCU is stored into the block pointed to by
632  * MCU_data[i]. WE ASSUME THIS AREA IS INITIALLY ZEROED BY THE CALLER.
633  * (Wholesale zeroing is usually a little faster than retail...)
634  *
635  * We return FALSE if data source requested suspension. In that case no
636  * changes have been made to permanent state. (Exception: some output
637  * coefficients may already have been assigned. This is harmless for
638  * spectral selection, since we'll just re-assign them on the next call.
639  * Successive approximation AC refinement has to be more careful, however.)
640  */
641 
642 /*
643  * MCU decoding for DC initial scan (either spectral selection,
644  * or first pass of successive approximation).
645  */
646 
647 METHODDEF(boolean)
648 decode_mcu_DC_first (j_decompress_ptr cinfo, JBLOCKROW *MCU_data)
649 {
650  huff_entropy_ptr entropy = (huff_entropy_ptr) cinfo->entropy;
651  int Al = cinfo->Al;
652  register int s, r;
653  int blkn, ci;
654  JBLOCKROW block;
655  BITREAD_STATE_VARS;
656  savable_state state;
657  d_derived_tbl * tbl;
658  jpeg_component_info * compptr;
659 
660  /* Process restart marker if needed; may have to suspend */
661  if (cinfo->restart_interval) {
662  if (entropy->restarts_to_go == 0)
663  if (! process_restart(cinfo))
664  return FALSE;
665  }
666 
667  /* If we've run out of data, just leave the MCU set to zeroes.
668  * This way, we return uniform gray for the remainder of the segment.
669  */
670  if (! entropy->pub.insufficient_data) {
671 
672  /* Load up working state */
673  BITREAD_LOAD_STATE(cinfo,entropy->bitstate);
674  ASSIGN_STATE(state, entropy->saved);
675 
676  /* Outer loop handles each block in the MCU */
677 
678  for (blkn = 0; blkn < cinfo->blocks_in_MCU; blkn++) {
679  block = MCU_data[blkn];
680  ci = cinfo->MCU_membership[blkn];
681  compptr = cinfo->cur_comp_info[ci];
682  tbl = entropy->derived_tbls[compptr->dc_tbl_no];
683 
684  /* Decode a single block's worth of coefficients */
685 
686  /* Section F.2.2.1: decode the DC coefficient difference */
687  HUFF_DECODE(s, br_state, tbl, return FALSE, label1);
688  if (s) {
689  CHECK_BIT_BUFFER(br_state, s, return FALSE);
690  r = GET_BITS(s);
691  s = HUFF_EXTEND(r, s);
692  }
693 
694  /* Convert DC difference to actual value, update last_dc_val */
695  s += state.last_dc_val[ci];
696  state.last_dc_val[ci] = s;
697  /* Scale and output the coefficient (assumes jpeg_natural_order[0]=0) */
698  (*block)[0] = (JCOEF) (s << Al);
699  }
700 
701  /* Completed MCU, so update state */
702  BITREAD_SAVE_STATE(cinfo,entropy->bitstate);
703  ASSIGN_STATE(entropy->saved, state);
704  }
705 
706  /* Account for restart interval (no-op if not using restarts) */
707  entropy->restarts_to_go--;
708 
709  return TRUE;
710 }

713 /*
714  * MCU decoding for AC initial scan (either spectral selection,
715  * or first pass of successive approximation).
716  */
717 
718 METHODDEF(boolean)
719 decode_mcu_AC_first (j_decompress_ptr cinfo, JBLOCKROW *MCU_data)
720 {
721  huff_entropy_ptr entropy = (huff_entropy_ptr) cinfo->entropy;
722  int Se = cinfo->Se;
723  int Al = cinfo->Al;
724  register int s, k, r;
725  unsigned int EOBRUN;
726  JBLOCKROW block;
727  BITREAD_STATE_VARS;
728  d_derived_tbl * tbl;
729 
730  /* Process restart marker if needed; may have to suspend */
731  if (cinfo->restart_interval) {
732  if (entropy->restarts_to_go == 0)
733  if (! process_restart(cinfo))
734  return FALSE;
735  }
736 
737  /* If we've run out of data, just leave the MCU set to zeroes.
738  * This way, we return uniform gray for the remainder of the segment.
739  */
740  if (! entropy->pub.insufficient_data) {
741 
742  /* Load up working state.
743  * We can avoid loading/saving bitread state if in an EOB run.
744  */
745  EOBRUN = entropy->saved.EOBRUN; /* only part of saved state we need */
746 
747  /* There is always only one block per MCU */
748 
749  if (EOBRUN > 0) /* if it's a band of zeroes... */
750  EOBRUN--; /* ...process it now (we do nothing) */
751  else {
752  BITREAD_LOAD_STATE(cinfo,entropy->bitstate);
753  block = MCU_data[0];
754  tbl = entropy->ac_derived_tbl;
755 
756  for (k = cinfo->Ss; k <= Se; k++) {
757  HUFF_DECODE(s, br_state, tbl, return FALSE, label2);
758  r = s >> 4;
759  s &= 15;
760  if (s) {
761  k += r;
762  CHECK_BIT_BUFFER(br_state, s, return FALSE);
763  r = GET_BITS(s);
764  s = HUFF_EXTEND(r, s);
765  /* Scale and output coefficient in natural (dezigzagged) order */
766  (*block)[jpeg_natural_order[k]] = (JCOEF) (s << Al);
767  } else {
768  if (r == 15) { /* ZRL */
769  k += 15; /* skip 15 zeroes in band */
770  } else { /* EOBr, run length is 2^r + appended bits */
771  EOBRUN = 1 << r;
772  if (r) { /* EOBr, r > 0 */
773  CHECK_BIT_BUFFER(br_state, r, return FALSE);
774  r = GET_BITS(r);
775  EOBRUN += r;
776  }
777  EOBRUN--; /* this band is processed at this moment */
778  break; /* force end-of-band */
779  }
780  }
781  }
782 
783  BITREAD_SAVE_STATE(cinfo,entropy->bitstate);
784  }
785 
786  /* Completed MCU, so update state */
787  entropy->saved.EOBRUN = EOBRUN; /* only part of saved state we need */
788  }
789 
790  /* Account for restart interval (no-op if not using restarts) */
791  entropy->restarts_to_go--;
792 
793  return TRUE;
794 }

803 METHODDEF(boolean)
804 decode_mcu_DC_refine (j_decompress_ptr cinfo, JBLOCKROW *MCU_data)
805 {
806  huff_entropy_ptr entropy = (huff_entropy_ptr) cinfo->entropy;
807  int p1 = 1 << cinfo->Al; /* 1 in the bit position being coded */
808  int blkn;
809  JBLOCKROW block;
810  BITREAD_STATE_VARS;
811 
812  /* Process restart marker if needed; may have to suspend */
813  if (cinfo->restart_interval) {
814  if (entropy->restarts_to_go == 0)
815  if (! process_restart(cinfo))
816  return FALSE;
817  }
818 
819  /* Not worth the cycles to check insufficient_data here,
820  * since we will not change the data anyway if we read zeroes.
821  */
822 
823  /* Load up working state */
824  BITREAD_LOAD_STATE(cinfo,entropy->bitstate);
825 
826  /* Outer loop handles each block in the MCU */
827

829  block = MCU_data[blkn];

832  CHECK_BIT_BUFFER(br_state, 1, return FALSE);
833  if (GET_BITS(1))
834  (*block)[0] |= p1;
835  /* Note: since we use |=, repeating the assignment later is safe */
836  }
837 
838  /* Completed MCU, so update state */
839  BITREAD_SAVE_STATE(cinfo,entropy->bitstate);
840 
841  /* Account for restart interval (no-op if not using restarts) */
842  entropy->restarts_to_go--;
843 
844  return TRUE;

848 /*
849  * MCU decoding for AC successive approximation refinement scan.
850  */
851 
852 METHODDEF(boolean)
853 decode_mcu_AC_refine (j_decompress_ptr cinfo, JBLOCKROW *MCU_data)
854 {
855  huff_entropy_ptr entropy = (huff_entropy_ptr) cinfo->entropy;
856  int Se = cinfo->Se;
857  int p1 = 1 << cinfo->Al; /* 1 in the bit position being coded */
858  int m1 = (-1) << cinfo->Al; /* -1 in the bit position being coded */
859  register int s, k, r;
860  unsigned int EOBRUN;
861  JBLOCKROW block;
862  JCOEFPTR thiscoef;
863  BITREAD_STATE_VARS;
864  d_derived_tbl * tbl;
865  int num_newnz;
866  int newnz_pos[DCTSIZE2];
867 
868  /* Process restart marker if needed; may have to suspend */
869  if (cinfo->restart_interval) {
870  if (entropy->restarts_to_go == 0)
871  if (! process_restart(cinfo))
872  return FALSE;
873  }
874 
875  /* If we've run out of data, don't modify the MCU.
876  */
877  if (! entropy->pub.insufficient_data) {
878 
879  /* Load up working state */
880  BITREAD_LOAD_STATE(cinfo,entropy->bitstate);
881  EOBRUN = entropy->saved.EOBRUN; /* only part of saved state we need */
882 
883  /* There is always only one block per MCU */
884  block = MCU_data[0];
885  tbl = entropy->ac_derived_tbl;
886 
887  /* If we are forced to suspend, we must undo the assignments to any newly
888  * nonzero coefficients in the block, because otherwise we'd get confused
889  * next time about which coefficients were already nonzero.
890  * But we need not undo addition of bits to already-nonzero coefficients;
891  * instead, we can test the current bit to see if we already did it.
892  */
893  num_newnz = 0;
894 
895  /* initialize coefficient loop counter to start of band */
896  k = cinfo->Ss;
897 
898  if (EOBRUN == 0) {
899  for (; k <= Se; k++) {
900  HUFF_DECODE(s, br_state, tbl, goto undoit, label3);
901  r = s >> 4;
902  s &= 15;
903  if (s) {
904  if (s != 1) /* size of new coef should always be 1 */
905  WARNMS(cinfo, JWRN_HUFF_BAD_CODE);
906  CHECK_BIT_BUFFER(br_state, 1, goto undoit);
907  if (GET_BITS(1))
908  s = p1; /* newly nonzero coef is positive */
909  else
910  s = m1; /* newly nonzero coef is negative */
911  } else {
912  if (r != 15) {
913  EOBRUN = 1 << r; /* EOBr, run length is 2^r + appended bits */
914  if (r) {
915  CHECK_BIT_BUFFER(br_state, r, goto undoit);
916  r = GET_BITS(r);
917  EOBRUN += r;
918  }
919  break; /* rest of block is handled by EOB logic */
920  }
921  /* note s = 0 for processing ZRL */
922  }
923  /* Advance over already-nonzero coefs and r still-zero coefs,
924  * appending correction bits to the nonzeroes. A correction bit is 1
925  * if the absolute value of the coefficient must be increased.
926  */
927  do {
928  thiscoef = *block + jpeg_natural_order[k];
929  if (*thiscoef != 0) {
930  CHECK_BIT_BUFFER(br_state, 1, goto undoit);
931  if (GET_BITS(1)) {
932  if ((*thiscoef & p1) == 0) { /* do nothing if already set it */
933  if (*thiscoef >= 0)
934  *thiscoef += p1;
935  else
936  *thiscoef += m1;
937  }
938  }
939  } else {
940  if (--r < 0)
941  break; /* reached target zero coefficient */
942  }
943  k++;
944  } while (k <= Se);
945  if (s) {
946  int pos = jpeg_natural_order[k];
947  /* Output newly nonzero coefficient */
948  (*block)[pos] = (JCOEF) s;
949  /* Remember its position in case we have to suspend */
950  newnz_pos[num_newnz++] = pos;
951  }
952  }
953  }
954 
955  if (EOBRUN > 0) {
956  /* Scan any remaining coefficient positions after the end-of-band
957  * (the last newly nonzero coefficient, if any). Append a correction
958  * bit to each already-nonzero coefficient. A correction bit is 1
959  * if the absolute value of the coefficient must be increased.
960  */
961  for (; k <= Se; k++) {
962  thiscoef = *block + jpeg_natural_order[k];
963  if (*thiscoef != 0) {
964  CHECK_BIT_BUFFER(br_state, 1, goto undoit);
965  if (GET_BITS(1)) {
966  if ((*thiscoef & p1) == 0) { /* do nothing if already changed it */
967  if (*thiscoef >= 0)
968  *thiscoef += p1;
969  else
970  *thiscoef += m1;
971  }
972  }
973  }
974  }
975  /* Count one block completed in EOB run */
976  EOBRUN--;
977  }
978 
979  /* Completed MCU, so update state */
980  BITREAD_SAVE_STATE(cinfo,entropy->bitstate);
981  entropy->saved.EOBRUN = EOBRUN; /* only part of saved state we need */
982  }
983 
984  /* Account for restart interval (no-op if not using restarts) */
985  entropy->restarts_to_go--;
986 
987  return TRUE;
988 
989 undoit:
990  /* Re-zero any output coefficients that we made newly nonzero */
991  while (num_newnz > 0)
992  (*block)[newnz_pos[--num_newnz]] = 0;
993 
994  return FALSE;
995 }

999  * Decode one MCU's worth of Huffman-compressed coefficients.
1000  */
1001 
1002 METHODDEF(boolean)
1003 decode_mcu (j_decompress_ptr cinfo, JBLOCKROW *MCU_data)
1004 {
1005  huff_entropy_ptr entropy = (huff_entropy_ptr) cinfo->entropy;
1006  int blkn;
1007  BITREAD_STATE_VARS;
1008  savable_state state;
1009 
1010  /* Process restart marker if needed; may have to suspend */
1011  if (cinfo->restart_interval) {
1012  if (entropy->restarts_to_go == 0)
1013  if (! process_restart(cinfo))
1014  return FALSE;
1015  }
1016 
1017  /* If we've run out of data, just leave the MCU set to zeroes.
1018  * This way, we return uniform gray for the remainder of the segment.
1019  */
1020  if (! entropy->pub.insufficient_data) {
1021 
1022  /* Load up working state */
1023  BITREAD_LOAD_STATE(cinfo,entropy->bitstate);
1024  ASSIGN_STATE(state, entropy->saved);
1025 
1026  /* Outer loop handles each block in the MCU */
1027 
1028  for (blkn = 0; blkn < cinfo->blocks_in_MCU; blkn++) {
1029  JBLOCKROW block = MCU_data[blkn];
1030  d_derived_tbl * htbl;
1031  register int s, k, r;
1032  int coef_limit, ci;
1033 
1034  /* Decode a single block's worth of coefficients */
1035 
1036  /* Section F.2.2.1: decode the DC coefficient difference */
1037  htbl = entropy->dc_cur_tbls[blkn];
1038  HUFF_DECODE(s, br_state, htbl, return FALSE, label1);
1039 
1040  htbl = entropy->ac_cur_tbls[blkn];
1041  k = 1;
1042  coef_limit = entropy->coef_limit[blkn];
1043  if (coef_limit) {
1044  /* Convert DC difference to actual value, update last_dc_val */
1045  if (s) {
1046  CHECK_BIT_BUFFER(br_state, s, return FALSE);
1047  r = GET_BITS(s);
1048  s = HUFF_EXTEND(r, s);
1049  }
1050  ci = cinfo->MCU_membership[blkn];
1051  s += state.last_dc_val[ci];
1052  state.last_dc_val[ci] = s;
1053  /* Output the DC coefficient */
1054  (*block)[0] = (JCOEF) s;
1055 
1056  /* Section F.2.2.2: decode the AC coefficients */
1057  /* Since zeroes are skipped, output area must be cleared beforehand */
1058  for (; k < coef_limit; k++) {
1059  HUFF_DECODE(s, br_state, htbl, return FALSE, label2);
1060 
1061  r = s >> 4;
1062  s &= 15;
1063 
1064  if (s) {
1065  k += r;
1066  CHECK_BIT_BUFFER(br_state, s, return FALSE);
1067  r = GET_BITS(s);
1068  s = HUFF_EXTEND(r, s);
1069  /* Output coefficient in natural (dezigzagged) order.
1070  * Note: the extra entries in jpeg_natural_order[] will save us
1071  * if k >= DCTSIZE2, which could happen if the data is corrupted.
1072  */
1073  (*block)[jpeg_natural_order[k]] = (JCOEF) s;
1074  } else {
1075  if (r != 15)
1076  goto EndOfBlock;
1077  k += 15;
1078  }
1079  }
1080  } else {
1081  if (s) {
1082  CHECK_BIT_BUFFER(br_state, s, return FALSE);
1083  DROP_BITS(s);
1084  }
1085  }
1086 
1087  /* Section F.2.2.2: decode the AC coefficients */
1088  /* In this path we just discard the values */
1089  for (; k < DCTSIZE2; k++) {
1090  HUFF_DECODE(s, br_state, htbl, return FALSE, label3);
1091 
1092  r = s >> 4;
1093  s &= 15;
1094 
1095  if (s) {
1096  k += r;
1097  CHECK_BIT_BUFFER(br_state, s, return FALSE);
1098  DROP_BITS(s);
1099  } else {
1100  if (r != 15)
1101  break;
1102  k += 15;
1103  }
1104  }
1105 
1106  EndOfBlock: ;
1107  }
1108 
1109  /* Completed MCU, so update state */
1110  BITREAD_SAVE_STATE(cinfo,entropy->bitstate);
1111  ASSIGN_STATE(entropy->saved, state);
1112  }
1113 
1114  /* Account for restart interval (no-op if not using restarts) */
1115  entropy->restarts_to_go--;
1116 
1117  return TRUE;
1118 }

1121 /*
1122  * Initialize for a Huffman-compressed scan.
1123  */
1124 
1125 METHODDEF(void)
1126 start_pass_huff_decoder (j_decompress_ptr cinfo)
1127 {
1128  huff_entropy_ptr entropy = (huff_entropy_ptr) cinfo->entropy;
1129  int ci, blkn, dctbl, actbl, i;
1130  jpeg_component_info * compptr;
1131 
1132  if (cinfo->progressive_mode) {
1133  /* Validate progressive scan parameters */
1134  if (cinfo->Ss == 0) {
1135  if (cinfo->Se != 0)
1136  goto bad;
1137  } else {
1138  /* need not check Ss/Se < 0 since they came from unsigned bytes */
1139  if (cinfo->Se < cinfo->Ss || cinfo->Se >= DCTSIZE2)
1140  goto bad;
1141  /* AC scans may have only one component */
1142  if (cinfo->comps_in_scan != 1)
1143  goto bad;
1144  }
1145  if (cinfo->Ah != 0) {
1146  /* Successive approximation refinement scan: must have Al = Ah-1. */
1147  if (cinfo->Ah-1 != cinfo->Al)
1148  goto bad;
1149  }
1150  if (cinfo->Al > 13) { /* need not check for < 0 */
1151  /* Arguably the maximum Al value should be less than 13 for 8-bit precision,
1152  * but the spec doesn't say so, and we try to be liberal about what we
1153  * accept. Note: large Al values could result in out-of-range DC
1154  * coefficients during early scans, leading to bizarre displays due to
1155  * overflows in the IDCT math. But we won't crash.
1156  */
1157  bad:
1158  ERREXIT4(cinfo, JERR_BAD_PROGRESSION,
1159  cinfo->Ss, cinfo->Se, cinfo->Ah, cinfo->Al);
1160  }
1161  /* Update progression status, and verify that scan order is legal.
1162  * Note that inter-scan inconsistencies are treated as warnings
1163  * not fatal errors ... not clear if this is right way to behave.
1164  */
1165  for (ci = 0; ci < cinfo->comps_in_scan; ci++) {
1166  int coefi, cindex = cinfo->cur_comp_info[ci]->component_index;
1167  int *coef_bit_ptr = & cinfo->coef_bits[cindex][0];
1168  if (cinfo->Ss && coef_bit_ptr[0] < 0) /* AC without prior DC scan */
1169  WARNMS2(cinfo, JWRN_BOGUS_PROGRESSION, cindex, 0);
1170  for (coefi = cinfo->Ss; coefi <= cinfo->Se; coefi++) {
1171  int expected = (coef_bit_ptr[coefi] < 0) ? 0 : coef_bit_ptr[coefi];
1172  if (cinfo->Ah != expected)
1173  WARNMS2(cinfo, JWRN_BOGUS_PROGRESSION, cindex, coefi);
1174  coef_bit_ptr[coefi] = cinfo->Al;
1175  }
1176  }
1177 
1178  /* Select MCU decoding routine */
1179  if (cinfo->Ah == 0) {
1180  if (cinfo->Ss == 0)
1181  entropy->pub.decode_mcu = decode_mcu_DC_first;
1182  else
1183  entropy->pub.decode_mcu = decode_mcu_AC_first;
1184  } else {
1185  if (cinfo->Ss == 0)
1186  entropy->pub.decode_mcu = decode_mcu_DC_refine;
1187  else
1188  entropy->pub.decode_mcu = decode_mcu_AC_refine;
1189  }
1190 
1191  for (ci = 0; ci < cinfo->comps_in_scan; ci++) {
1192  compptr = cinfo->cur_comp_info[ci];
1193  /* Make sure requested tables are present, and compute derived tables.
1194  * We may build same derived table more than once, but it's not expensive.
1195  */
1196  if (cinfo->Ss == 0) {
1197  if (cinfo->Ah == 0) { /* DC refinement needs no table */
1198  i = compptr->dc_tbl_no;
1199  jpeg_make_d_derived_tbl(cinfo, TRUE, i,
1200  & entropy->derived_tbls[i]);
1201  }
1202  } else {
1203  i = compptr->ac_tbl_no;
1204  jpeg_make_d_derived_tbl(cinfo, FALSE, i,
1205  & entropy->derived_tbls[i]);
1206  /* remember the single active table */
1207  entropy->ac_derived_tbl = entropy->derived_tbls[i];
