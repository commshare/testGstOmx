343 void omx_videoenc_component_BufferMgmtCallback(OMX_COMPONENTTYPE *openmaxStandComp, OMX_BUFFERHEADERTYPE* pInputBuffer, OMX_BUFFERHEADERTYPE* pOutputBuffer) {
344 
345  omx_videoenc_component_PrivateType* omx_videoenc_component_Private = openmaxStandComp->pComponentPrivate;
346  omx_base_video_PortType *inPort = (omx_base_video_PortType *)omx_videoenc_component_Private->ports[OMX_BASE_FILTER_INPUTPORT_INDEX];
347 
348  OMX_S32 nOutputFilled = 0;
349  OMX_U8* outputCurrBuffer;
350  OMX_S32 nLen = 0;
351  int size;
352 
353  size= inPort->sPortParam.format.video.nFrameWidth*inPort->sPortParam.format.video.nFrameHeight;
354 
355  DEBUG(DEB_LEV_FUNCTION_NAME, "In %s\n", __func__);
357  if(omx_videoenc_component_Private->isNewBuffer) {
358  omx_videoenc_component_Private->isNewBuffer = 0;
359  DEBUG(DEB_LEV_FULL_SEQ, "New Buffer FilledLen = %d\n", (int)pInputBuffer->nFilledLen);
360 
361  omx_videoenc_component_Private->picture->data[0] = pInputBuffer->pBuffer;
362  omx_videoenc_component_Private->picture->data[1] = omx_videoenc_component_Private->picture->data[0] + size;
363  omx_videoenc_component_Private->picture->data[2] = omx_videoenc_component_Private->picture->data[1] + size / 4;
364  omx_videoenc_component_Private->picture->linesize[0] = inPort->sPortParam.format.video.nFrameWidth;
365  omx_videoenc_component_Private->picture->linesize[1] = inPort->sPortParam.format.video.nFrameWidth / 2;
366  omx_videoenc_component_Private->picture->linesize[2] = inPort->sPortParam.format.video.nFrameWidth / 2;
367  }
368 
369  outputCurrBuffer = pOutputBuffer->pBuffer;
370  pOutputBuffer->nFilledLen = 0;
371  pOutputBuffer->nOffset = 0;
372 
373  while (!nOutputFilled) {
374  if (omx_videoenc_component_Private->isFirstBuffer) {
375  tsem_down(omx_videoenc_component_Private->avCodecSyncSem);
376  omx_videoenc_component_Private->isFirstBuffer = 0;
377  }
378  omx_videoenc_component_Private->avCodecContext->frame_number++;
379 
380  nLen = avcodec_encode_video(omx_videoenc_component_Private->avCodecContext,
381  outputCurrBuffer,
382  pOutputBuffer->nAllocLen,
383  omx_videoenc_component_Private->picture);
384 
385  if (nLen < 0) {
386  DEBUG(DEB_LEV_ERR, "----> A general error or simply frame not encoded?\n");
387  }
388 
389  pInputBuffer->nFilledLen = 0;
390  omx_videoenc_component_Private->isNewBuffer = 1;
391  if ( nLen >= 0) {
392  pOutputBuffer->nFilledLen = nLen;
393  }
394  nOutputFilled = 1;
395  }
396  DEBUG(DEB_LEV_FULL_SEQ, "One output buffer %x nLen=%d is full returning in video encoder\n",
397  (int)pOutputBuffer->pBuffer, (int)pOutputBuffer->nFilledLen);
398 }
