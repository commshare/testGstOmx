280 
281  if (hOutput == NULL && hInput == NULL)
282  return OMX_ErrorBadParameter;

284  err = (component->ComponentTunnelRequest)(hOutput, nPortOutput, hInput, nPortInput, tunnelSetup);
285  if (err != OMX_ErrorNone) {

290  }

295 
296  component = (OMX_COMPONENTTYPE*)hInput;
297  if (hInput) {
298  err = (component->ComponentTunnelRequest)(hInput, nPortInput, hOutput, nPortOutput, tunnelSetup);
299  if (err != OMX_ErrorNone) {
300  DEBUG(DEB_LEV_ERR, "Tunneling failed: input port rejects it - err = %08x\n", err);
301  // the second stage fails. the tunnel on poutput port has to be removed
302  component = (OMX_COMPONENTTYPE*)hOutput;
303  err = (component->ComponentTunnelRequest)(hOutput, nPortOutput, NULL, 0, tunnelSetup);
304  if (err != OMX_ErrorNone) {
305  // This error should never happen. It is critical, and not recoverable
306  free(tunnelSetup);
307  tunnelSetup = NULL;
308  DEBUG(DEB_LEV_FUNCTION_NAME, "Out of %s with OMX_ErrorUndefined\n", __func__);
309  return OMX_ErrorUndefined;
310  }
311  free(tunnelSetup);
312  tunnelSetup = NULL;
313  DEBUG(DEB_LEV_FUNCTION_NAME, "Out of %s with OMX_ErrorPortsNotCompatible\n", __func__);
314  return OMX_ErrorPortsNotCompatible;
315  }
316  }
317  DEBUG(DEB_LEV_PARAMS, "Second stage of tunneling acheived:\n");
318  DEBUG(DEB_LEV_PARAMS, " - supplier proposed = %i\n", (int)tunnelSetup->eSupplier);
319  DEBUG(DEB_LEV_PARAMS, " - flags = %i\n", (int)tunnelSetup->nTunnelFlags);
320  free(tunnelSetup);
321  tunnelSetup = NULL;
322  DEBUG(DEB_LEV_FUNCTION_NAME, "Out of %s\n", __func__);
323  return OMX_ErrorNone;
