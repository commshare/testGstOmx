335 static inline int scale_int (mad_fixed_t sample) {
336 
337  #if MAD_F_FRACBITS < 28
338  /* round */
339  sample += (1L << (28 - MAD_F_FRACBITS - 1));
340  #endif
341 
342  /* clip */
343  if (sample >= MAD_F_ONE)
344  sample = MAD_F_ONE - 1;
345  else if (sample < -MAD_F_ONE)
346  sample = -MAD_F_ONE;
347 
348  #if MAD_F_FRACBITS < 28
349  /* quantize */
350  sample >>= (28 - MAD_F_FRACBITS);
351  #endif
352 
353  /* convert from 29 bits to 32 bits */
354  return (int) (sample << 3);
355 }
