420 static int print_text_marker (j_decompress_ptr cinfo)
421 {
422  boolean traceit = (cinfo->err->trace_level >= 1);
423  INT32 length;
424  OMX_U32 ch;
425  OMX_U32 lastch = 0;
426 
427  length = jpeg_getc(cinfo) << 8;
428  length += jpeg_getc(cinfo);
429  length -= 2; /* discount the length word itself */
430 
431  if (traceit) {
432  if (cinfo->unread_marker == JPEG_COM)
433  DEBUG(DEB_LEV_PARAMS, "Comment, length %ld:\n", (long) length);
434  else /* assume it is an APPn otherwise */
435  DEBUG(DEB_LEV_PARAMS, "APP%d, length %ld:\n",
436  cinfo->unread_marker - JPEG_APP0, (long) length);
437  }
438 
439  while (--length >= 0) {
440  ch = jpeg_getc(cinfo);
441  if (traceit) {
442  /* Emit the character in a readable form.
443  * Nonprintables are converted to \nnn form,
444  * while \ is converted to \\.
445  * Newlines in CR, CR/LF, or LF form will be printed as one newline.
446  */
447  if (ch == '\r') {
448  DEBUG(DEB_LEV_PARAMS, "\n");
449  } else if (ch == '\n') {
450  if (lastch != '\r')
451  DEBUG(DEB_LEV_PARAMS, "\n");
452  } else if (ch == '\\') {
453  DEBUG(DEB_LEV_PARAMS, "\\\\");
454  } else if (isprint(ch)) {
455  putc(ch, stderr);
456  } else {
457  DEBUG(DEB_LEV_PARAMS, "\\%03o", (unsigned int)ch);
458  }
459  lastch = ch;
460  }
461  }
462 
463  if (traceit)
464  DEBUG(DEB_LEV_PARAMS, "\n");
465 
466  return TRUE;
