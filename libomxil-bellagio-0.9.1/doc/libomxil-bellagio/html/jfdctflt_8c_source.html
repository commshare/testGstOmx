6  * For conditions of distribution and use, see the accompanying README file.
7  *
8  * This file contains a floating-point implementation of the
9  * forward DCT (Discrete Cosine Transform).
10  *
11  * This implementation should be more accurate than either of the integer
12  * DCT implementations. However, it may not give the same results on all
13  * machines because of differences in roundoff behavior. Speed will depend
14  * on the hardware's floating point capacity.
15  *
16  * A 2-D DCT can be done by 1-D DCT on each row followed by 1-D DCT
17  * on each column. Direct algorithms are also available, but they are
18  * much more complex and seem not to be any faster when reduced to code.
19  *
20  * This implementation is based on Arai, Agui, and Nakajima's algorithm for
21  * scaled DCT. Their original paper (Trans. IEICE E-71(11):1095) is in
22  * Japanese, but the algorithm is described in the Pennebaker & Mitchell
23  * JPEG textbook (see REFERENCES section in file README). The following code
24  * is based directly on figure 4-8 in P&M.
25  * While an 8-point DCT cannot be done in less than 11 multiplies, it is
26  * possible to arrange the computation so that many of the multiplies are
27  * simple scalings of the final outputs. These multiplies can then be
28  * folded into the multiplications or divisions by the JPEG quantization
29  * table entries. The AA&N method leaves only 5 multiplies and 29 adds
30  * to be done in the DCT itself.
31  * The primary disadvantage of this method is that with a fixed-point
32  * implementation, accuracy is lost due to imprecise representation of the
33  * scaled quantization values. However, that problem does not arise if
34  * we use floating point arithmetic.
35  */

68  /* Pass 1: process rows. */

71  for (ctr = 0; ctr < DCTSIZE; ctr++) {
72  elemptr = sample_data[ctr] + start_col;
73 
74  /* Load data into workspace, applying unsigned->signed conversion */
75  tmp0 = (FAST_FLOAT) (GETJSAMPLE(elemptr[0]) + GETJSAMPLE(elemptr[7]));
76  tmp7 = (FAST_FLOAT) (GETJSAMPLE(elemptr[0]) - GETJSAMPLE(elemptr[7]));
77  tmp1 = (FAST_FLOAT) (GETJSAMPLE(elemptr[1]) + GETJSAMPLE(elemptr[6]));
78  tmp6 = (FAST_FLOAT) (GETJSAMPLE(elemptr[1]) - GETJSAMPLE(elemptr[6]));
79  tmp2 = (FAST_FLOAT) (GETJSAMPLE(elemptr[2]) + GETJSAMPLE(elemptr[5]));
80  tmp5 = (FAST_FLOAT) (GETJSAMPLE(elemptr[2]) - GETJSAMPLE(elemptr[5]));
81  tmp3 = (FAST_FLOAT) (GETJSAMPLE(elemptr[3]) + GETJSAMPLE(elemptr[4]));
82  tmp4 = (FAST_FLOAT) (GETJSAMPLE(elemptr[3]) - GETJSAMPLE(elemptr[4]));
83 
84  /* Even part */
85 
86  tmp10 = tmp0 + tmp3; /* phase 2 */
87  tmp13 = tmp0 - tmp3;
88  tmp11 = tmp1 + tmp2;
89  tmp12 = tmp1 - tmp2;
90 
91  /* Apply unsigned->signed conversion */
92  dataptr[0] = tmp10 + tmp11 - 8 * CENTERJSAMPLE; /* phase 3 */
93  dataptr[4] = tmp10 - tmp11;
94 
95  z1 = (tmp12 + tmp13) * ((FAST_FLOAT) 0.707106781); /* c4 */
96  dataptr[2] = tmp13 + z1; /* phase 5 */
97  dataptr[6] = tmp13 - z1;
98 
99  /* Odd part */
100 
101  tmp10 = tmp4 + tmp5; /* phase 2 */
102  tmp11 = tmp5 + tmp6;
103  tmp12 = tmp6 + tmp7;
104 
105  /* The rotator is modified from fig 4-8 to avoid extra negations. */
106  z5 = (tmp10 - tmp12) * ((FAST_FLOAT) 0.382683433); /* c6 */
107  z2 = ((FAST_FLOAT) 0.541196100) * tmp10 + z5; /* c2-c6 */
108  z4 = ((FAST_FLOAT) 1.306562965) * tmp12 + z5; /* c2+c6 */
109  z3 = tmp11 * ((FAST_FLOAT) 0.707106781); /* c4 */
110

112  z13 = tmp7 - z3;
113

115  dataptr[3] = z13 - z2;
116  dataptr[1] = z11 + z4;
117  dataptr[7] = z11 - z4;
118 
119  dataptr += DCTSIZE; /* advance pointer to next row */
120  }
121 
122  /* Pass 2: process columns. */
123

125  for (ctr = DCTSIZE-1; ctr >= 0; ctr--) {
126  tmp0 = dataptr[DCTSIZE*0] + dataptr[DCTSIZE*7];
127  tmp7 = dataptr[DCTSIZE*0] - dataptr[DCTSIZE*7];
128  tmp1 = dataptr[DCTSIZE*1] + dataptr[DCTSIZE*6];
129  tmp6 = dataptr[DCTSIZE*1] - dataptr[DCTSIZE*6];
130  tmp2 = dataptr[DCTSIZE*2] + dataptr[DCTSIZE*5];
131  tmp5 = dataptr[DCTSIZE*2] - dataptr[DCTSIZE*5];
132  tmp3 = dataptr[DCTSIZE*3] + dataptr[DCTSIZE*4];
133  tmp4 = dataptr[DCTSIZE*3] - dataptr[DCTSIZE*4];
134 
135  /* Even part */
136 
137  tmp10 = tmp0 + tmp3; /* phase 2 */
138  tmp13 = tmp0 - tmp3;
139  tmp11 = tmp1 + tmp2;
140  tmp12 = tmp1 - tmp2;
141 
142  dataptr[DCTSIZE*0] = tmp10 + tmp11; /* phase 3 */
143  dataptr[DCTSIZE*4] = tmp10 - tmp11;
144 
145  z1 = (tmp12 + tmp13) * ((FAST_FLOAT) 0.707106781); /* c4 */
146  dataptr[DCTSIZE*2] = tmp13 + z1; /* phase 5 */
147  dataptr[DCTSIZE*6] = tmp13 - z1;
148 
149  /* Odd part */
150 
151  tmp10 = tmp4 + tmp5; /* phase 2 */
152  tmp11 = tmp5 + tmp6;
153  tmp12 = tmp6 + tmp7;
154 
155  /* The rotator is modified from fig 4-8 to avoid extra negations. */
156  z5 = (tmp10 - tmp12) * ((FAST_FLOAT) 0.382683433); /* c6 */
157  z2 = ((FAST_FLOAT) 0.541196100) * tmp10 + z5; /* c2-c6 */
158  z4 = ((FAST_FLOAT) 1.306562965) * tmp12 + z5; /* c2+c6 */

163

165  dataptr[DCTSIZE*3] = z13 - z2;
166  dataptr[DCTSIZE*1] = z11 + z4;
167  dataptr[DCTSIZE*7] = z11 - z4;
