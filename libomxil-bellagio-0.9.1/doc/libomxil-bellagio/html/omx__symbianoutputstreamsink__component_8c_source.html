305 {
306  OMX_AUDIO_PARAM_PORTFORMATTYPE *pAudioPortFormat;
307 
308  OMX_COMPONENTTYPE *openmaxStandComp = (OMX_COMPONENTTYPE*)hComponent;

310  omx_symbianoutputstreamsink_component_PortType *pPort = (omx_symbianoutputstreamsink_component_PortType *) omx_symbianoutputstreamsink_component_Private->ports[OMX_BASE_SINK_INPUTPORT_INDEX];
311 
312  if (ComponentParameterStructure == NULL)
313  {
314  return OMX_ErrorBadParameter;
315  }
316 
317  DEBUG(DEB_LEV_SIMPLE_SEQ, " Getting parameter %i\n", nParamIndex);
318 
319  /* Check which structure we are being fed and fill its header */
320  switch(nParamIndex)
321  {
322  case OMX_IndexParamAudioInit:
323  setHeader(ComponentParameterStructure, sizeof(OMX_PORT_PARAM_TYPE));
324  memcpy(ComponentParameterStructure, &omx_symbianoutputstreamsink_component_Private->sPortTypesParam[OMX_PortDomainAudio], sizeof(OMX_PORT_PARAM_TYPE));
325  break;
326 
327  case OMX_IndexParamAudioPortFormat:
328  pAudioPortFormat = (OMX_AUDIO_PARAM_PORTFORMATTYPE*)ComponentParameterStructure;
329  setHeader(pAudioPortFormat, sizeof(OMX_AUDIO_PARAM_PORTFORMATTYPE));
330  if (pAudioPortFormat->nPortIndex < 1)
331  {
332  memcpy(pAudioPortFormat, &pPort->sAudioParam, sizeof(OMX_AUDIO_PARAM_PORTFORMATTYPE));
333  }
334  else
335  {
336  return OMX_ErrorBadPortIndex;
337  }
338  break;
339 
340  case OMX_IndexParamAudioPcm:
341  if(((OMX_AUDIO_PARAM_PCMMODETYPE*)ComponentParameterStructure)->nPortIndex !=
342  pPort->omxAudioParamPcmMode.nPortIndex)
343  {
344  return OMX_ErrorBadParameter;
345  }
346  memcpy(ComponentParameterStructure, &pPort->omxAudioParamPcmMode, sizeof(OMX_AUDIO_PARAM_PCMMODETYPE));
347  break;
348 
349  default: /*Call the base component function*/
350  return omx_base_component_GetParameter(hComponent, nParamIndex, ComponentParameterStructure);
351  }
352 
353  return OMX_ErrorNone;
354 }
