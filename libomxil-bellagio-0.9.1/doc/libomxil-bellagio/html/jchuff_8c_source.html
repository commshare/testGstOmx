166  * Compute the derived values for a Huffman table.
167  * This routine also performs some validation checks on the table.
168  */
169 
170 LOCAL(void)
171 jpeg_make_c_derived_tbl (j_compress_ptr cinfo, boolean isDC, int tblno,
172  c_derived_tbl ** pdtbl)
173 {
174  JHUFF_TBL *htbl;
175  c_derived_tbl *dtbl;
176  int p, i, l, lastp, si, maxsymbol;
177  char huffsize[257];
178  unsigned int huffcode[257];
179  unsigned int code;
180 
181  /* Note that huffsize[] and huffcode[] are filled in code-length order,
182  * paralleling the order of the symbols themselves in htbl->huffval[].
183  */
184 
185  /* Find the input Huffman table */
186  if (tblno < 0 || tblno >= NUM_HUFF_TBLS)
187  ERREXIT1(cinfo, JERR_NO_HUFF_TABLE, tblno);
188  htbl =
189  isDC ? cinfo->dc_huff_tbl_ptrs[tblno] : cinfo->ac_huff_tbl_ptrs[tblno];
190  if (htbl == NULL)
191  ERREXIT1(cinfo, JERR_NO_HUFF_TABLE, tblno);
192 
193  /* Allocate a workspace if we haven't already done so. */
194  if (*pdtbl == NULL)
195  *pdtbl = (c_derived_tbl *)
196  (*cinfo->mem->alloc_small) ((j_common_ptr) cinfo, JPOOL_IMAGE,
197  SIZEOF(c_derived_tbl));
198  dtbl = *pdtbl;
199 
200  /* Figure C.1: make table of Huffman code length for each symbol */
201 
202  p = 0;
203  for (l = 1; l <= 16; l++) {
204  i = (int) htbl->bits[l];
205  if (i < 0 || p + i > 256) /* protect against table overrun */
206  ERREXIT(cinfo, JERR_BAD_HUFF_TABLE);
207  while (i--)
208  huffsize[p++] = (char) l;
209  }
210  huffsize[p] = 0;
211  lastp = p;
212 
213  /* Figure C.2: generate the codes themselves */
214  /* We also validate that the counts represent a legal Huffman code tree. */
215 
216  code = 0;
217  si = huffsize[0];
218  p = 0;
219  while (huffsize[p]) {
220  while (((int) huffsize[p]) == si) {
221  huffcode[p++] = code;
222  code++;
223  }
224  /* code is now 1 more than the last code used for codelength si; but
225  * it must still fit in si bits, since no code is allowed to be all ones.
226  */
227  if (((INT32) code) >= (((INT32) 1) << si))
228  ERREXIT(cinfo, JERR_BAD_HUFF_TABLE);
229  code <<= 1;
230  si++;
231  }
232 
233  /* Figure C.3: generate encoding tables */
234  /* These are code and size indexed by symbol value */
235 
236  /* Set all codeless symbols to have code length 0;
237  * this lets us detect duplicate VAL entries here, and later
238  * allows emit_bits to detect any attempt to emit such symbols.
239  */
240  MEMZERO(dtbl->ehufsi, SIZEOF(dtbl->ehufsi));
241 
242  /* This is also a convenient place to check for out-of-range
243  * and duplicated VAL entries. We allow 0..255 for AC symbols
244  * but only 0..15 for DC. (We could constrain them further
245  * based on data depth and mode, but this seems enough.)
246  */
247  maxsymbol = isDC ? 15 : 255;
248 
249  for (p = 0; p < lastp; p++) {
250  i = htbl->huffval[p];
251  if (i < 0 || i > maxsymbol || dtbl->ehufsi[i])
252  ERREXIT(cinfo, JERR_BAD_HUFF_TABLE);
253  dtbl->ehufco[i] = huffcode[p];
254  dtbl->ehufsi[i] = huffsize[p];
255  }
256 }

309 /* Only the right 24 bits of put_buffer are used; the valid bits are
310  * left-justified in this part. At most 16 bits can be passed to emit_bits

312  * between calls, so 24 bits are sufficient.

315 INLINE
316 LOCAL(boolean)
317 emit_bits_s (working_state * state, unsigned int code, int size)
318 /* Emit some bits; return TRUE if successful, FALSE if must suspend */
319 {
320  /* This routine is heavily used, so it's worth coding tightly. */
321  register INT32 put_buffer = (INT32) code;
322  register int put_bits = state->cur.put_bits;
323 
324  /* if size is 0, caller used an invalid Huffman table entry */
325  if (size == 0)
326  ERREXIT(state->cinfo, JERR_HUFF_MISSING_CODE);
327 
328  put_buffer &= (((INT32) 1)<<size) - 1; /* mask off any extra bits in code */
329 
330  put_bits += size; /* new number of bits in buffer */
331 
332  put_buffer <<= 24 - put_bits; /* align incoming bits */
333 
334  put_buffer |= state->cur.put_buffer; /* and merge with old buffer contents */
335 
336  while (put_bits >= 8) {
337  int c = (int) ((put_buffer >> 16) & 0xFF);
338 
339  emit_byte_s(state, c, return FALSE);
340  if (c == 0xFF) { /* need to stuff a zero byte? */
341  emit_byte_s(state, 0, return FALSE);
342  }
343  put_buffer <<= 8;
344  put_bits -= 8;
345  }
346 
347  state->cur.put_buffer = put_buffer; /* update state variables */
348  state->cur.put_bits = put_bits;
349 
350  return TRUE;
351 }

537 METHODDEF(boolean)
538 encode_mcu_DC_first (j_compress_ptr cinfo, JBLOCKROW *MCU_data)
539 {
540  huff_entropy_ptr entropy = (huff_entropy_ptr) cinfo->entropy;
541  register int temp, temp2;
542  register int nbits;
543  int blkn, ci;
544  int Al = cinfo->Al;
545  JBLOCKROW block;
546  jpeg_component_info * compptr;
547  ISHIFT_TEMPS
548 
549  entropy->next_output_byte = cinfo->dest->next_output_byte;
550  entropy->free_in_buffer = cinfo->dest->free_in_buffer;
551 
552  /* Emit restart marker if needed */
553  if (cinfo->restart_interval)
554  if (entropy->restarts_to_go == 0)
555  emit_restart_e(entropy, entropy->next_restart_num);
556 
557  /* Encode the MCU data blocks */
558  for (blkn = 0; blkn < cinfo->blocks_in_MCU; blkn++) {
559  block = MCU_data[blkn];
560  ci = cinfo->MCU_membership[blkn];
561  compptr = cinfo->cur_comp_info[ci];
562 
563  /* Compute the DC value after the required point transform by Al.
564  * This is simply an arithmetic right shift.
565  */
566  temp2 = IRIGHT_SHIFT((int) ((*block)[0]), Al);
567 
568  /* DC differences are figured on the point-transformed values. */
569  temp = temp2 - entropy->saved.last_dc_val[ci];
570  entropy->saved.last_dc_val[ci] = temp2;
571 
572  /* Encode the DC coefficient difference per section G.1.2.1 */
573  temp2 = temp;
574  if (temp < 0) {
575  temp = -temp; /* temp is abs value of input */
576  /* For a negative input, want temp2 = bitwise complement of abs(input) */
577  /* This code assumes we are on a two's complement machine */
578  temp2--;
579  }
580 
581  /* Find the number of bits needed for the magnitude of the coefficient */
582  nbits = 0;
583  while (temp) {
584  nbits++;
585  temp >>= 1;
586  }
587  /* Check for out-of-range coefficient values.
588  * Since we're encoding a difference, the range limit is twice as much.
589  */
590  if (nbits > MAX_COEF_BITS+1)
591  ERREXIT(cinfo, JERR_BAD_DCT_COEF);
592 
593  /* Count/emit the Huffman-coded symbol for the number of bits */
594  emit_symbol(entropy, compptr->dc_tbl_no, nbits);
595 
596  /* Emit that number of bits of the value, if positive, */
597  /* or the complement of its magnitude, if negative. */
598  if (nbits) /* emit_bits rejects calls with size 0 */
599  emit_bits_e(entropy, (unsigned int) temp2, nbits);
600  }
601 
602  cinfo->dest->next_output_byte = entropy->next_output_byte;
603  cinfo->dest->free_in_buffer = entropy->free_in_buffer;
604 
605  /* Update restart-interval state too */
606  if (cinfo->restart_interval) {
607  if (entropy->restarts_to_go == 0) {
608  entropy->restarts_to_go = cinfo->restart_interval;
609  entropy->next_restart_num++;
610  entropy->next_restart_num &= 7;
611  }
612  entropy->restarts_to_go--;
613  }
614 
615  return TRUE;
616 }

624 METHODDEF(boolean)
625 encode_mcu_AC_first (j_compress_ptr cinfo, JBLOCKROW *MCU_data)
626 {
627  huff_entropy_ptr entropy = (huff_entropy_ptr) cinfo->entropy;
628  register int temp, temp2;
629  register int nbits;
630  register int r, k;
631  int Se = cinfo->Se;
632  int Al = cinfo->Al;
633  JBLOCKROW block;
634 
635  entropy->next_output_byte = cinfo->dest->next_output_byte;
636  entropy->free_in_buffer = cinfo->dest->free_in_buffer;
637 
638  /* Emit restart marker if needed */
639  if (cinfo->restart_interval)
640  if (entropy->restarts_to_go == 0)
641  emit_restart_e(entropy, entropy->next_restart_num);
642 
643  /* Encode the MCU data block */
644  block = MCU_data[0];
645 
646  /* Encode the AC coefficients per section G.1.2.2, fig. G.3 */
647 
648  r = 0; /* r = run length of zeros */
649 
650  for (k = cinfo->Ss; k <= Se; k++) {
651  if ((temp = (*block)[jpeg_natural_order[k]]) == 0) {
652  r++;
653  continue;
654  }
655  /* We must apply the point transform by Al. For AC coefficients this
656  * is an integer division with rounding towards 0. To do this portably
657  * in C, we shift after obtaining the absolute value; so the code is
658  * interwoven with finding the abs value (temp) and output bits (temp2).
659  */
660  if (temp < 0) {
661  temp = -temp; /* temp is abs value of input */
662  temp >>= Al; /* apply the point transform */
663  /* For a negative coef, want temp2 = bitwise complement of abs(coef) */
664  temp2 = ~temp;
665  } else {
666  temp >>= Al; /* apply the point transform */
667  temp2 = temp;
668  }
669  /* Watch out for case that nonzero coef is zero after point transform */
670  if (temp == 0) {
671  r++;
672  continue;
673  }
674 
675  /* Emit any pending EOBRUN */
676  if (entropy->EOBRUN > 0)
677  emit_eobrun(entropy);
678  /* if run length > 15, must emit special run-length-16 codes (0xF0) */
679  while (r > 15) {
680  emit_symbol(entropy, entropy->ac_tbl_no, 0xF0);
681  r -= 16;
682  }
683 
684  /* Find the number of bits needed for the magnitude of the coefficient */
685  nbits = 1; /* there must be at least one 1 bit */
686  while ((temp >>= 1))
687  nbits++;
688  /* Check for out-of-range coefficient values */
689  if (nbits > MAX_COEF_BITS)
690  ERREXIT(cinfo, JERR_BAD_DCT_COEF);
691 
692  /* Count/emit Huffman symbol for run length / number of bits */
693  emit_symbol(entropy, entropy->ac_tbl_no, (r << 4) + nbits);
694 
695  /* Emit that number of bits of the value, if positive, */
696  /* or the complement of its magnitude, if negative. */
697  emit_bits_e(entropy, (unsigned int) temp2, nbits);
698 
699  r = 0; /* reset zero run length */
700  }
701 
702  if (r > 0) { /* If there are trailing zeroes, */
703  entropy->EOBRUN++; /* count an EOB */
704  if (entropy->EOBRUN == 0x7FFF)
705  emit_eobrun(entropy); /* force it out to avoid overflow */
706  }
707 
708  cinfo->dest->next_output_byte = entropy->next_output_byte;
709  cinfo->dest->free_in_buffer = entropy->free_in_buffer;
710 
711  /* Update restart-interval state too */
712  if (cinfo->restart_interval) {
713  if (entropy->restarts_to_go == 0) {
714  entropy->restarts_to_go = cinfo->restart_interval;
715  entropy->next_restart_num++;
716  entropy->next_restart_num &= 7;
717  }
718  entropy->restarts_to_go--;
719  }
720 
721  return TRUE;
722 }

778 METHODDEF(boolean)
779 encode_mcu_AC_refine (j_compress_ptr cinfo, JBLOCKROW *MCU_data)
780 {
781  huff_entropy_ptr entropy = (huff_entropy_ptr) cinfo->entropy;
782  register int temp;
783  register int r, k;
784  int EOB;
785  char *BR_buffer;
786  unsigned int BR;
787  int Se = cinfo->Se;
788  int Al = cinfo->Al;
789  JBLOCKROW block;
790  int absvalues[DCTSIZE2];
791 
792  entropy->next_output_byte = cinfo->dest->next_output_byte;
793  entropy->free_in_buffer = cinfo->dest->free_in_buffer;
794 
795  /* Emit restart marker if needed */
796  if (cinfo->restart_interval)
797  if (entropy->restarts_to_go == 0)
798  emit_restart_e(entropy, entropy->next_restart_num);
799 
800  /* Encode the MCU data block */
801  block = MCU_data[0];
802 
803  /* It is convenient to make a pre-pass to determine the transformed
804  * coefficients' absolute values and the EOB position.
805  */
806  EOB = 0;
807  for (k = cinfo->Ss; k <= Se; k++) {
808  temp = (*block)[jpeg_natural_order[k]];
809  /* We must apply the point transform by Al. For AC coefficients this
810  * is an integer division with rounding towards 0. To do this portably
811  * in C, we shift after obtaining the absolute value.
812  */
813  if (temp < 0)
814  temp = -temp; /* temp is abs value of input */
815  temp >>= Al; /* apply the point transform */
816  absvalues[k] = temp; /* save abs value for main pass */
817  if (temp == 1)
818  EOB = k; /* EOB = index of last newly-nonzero coef */
819  }
820 
821  /* Encode the AC coefficients per section G.1.2.3, fig. G.7 */
822 
823  r = 0; /* r = run length of zeros */
824  BR = 0; /* BR = count of buffered bits added now */
825  BR_buffer = entropy->bit_buffer + entropy->BE; /* Append bits to buffer */
826 
827  for (k = cinfo->Ss; k <= Se; k++) {
828  if ((temp = absvalues[k]) == 0) {
829  r++;
830  continue;
831  }
832 
833  /* Emit any required ZRLs, but not if they can be folded into EOB */
834  while (r > 15 && k <= EOB) {
835  /* emit any pending EOBRUN and the BE correction bits */
836  emit_eobrun(entropy);
837  /* Emit ZRL */
838  emit_symbol(entropy, entropy->ac_tbl_no, 0xF0);
839  r -= 16;
840  /* Emit buffered correction bits that must be associated with ZRL */
841  emit_buffered_bits(entropy, BR_buffer, BR);
842  BR_buffer = entropy->bit_buffer; /* BE bits are gone now */
843  BR = 0;
844  }
845 
846  /* If the coef was previously nonzero, it only needs a correction bit.
847  * NOTE: a straight translation of the spec's figure G.7 would suggest
848  * that we also need to test r > 15. But if r > 15, we can only get here
849  * if k > EOB, which implies that this coefficient is not 1.
850  */
851  if (temp > 1) {
852  /* The correction bit is the next bit of the absolute value. */
853  BR_buffer[BR++] = (char) (temp & 1);
854  continue;
855  }
856 
857  /* Emit any pending EOBRUN and the BE correction bits */
858  emit_eobrun(entropy);
859 
860  /* Count/emit Huffman symbol for run length / number of bits */
861  emit_symbol(entropy, entropy->ac_tbl_no, (r << 4) + 1);
862 
863  /* Emit output bit for newly-nonzero coef */
864  temp = ((*block)[jpeg_natural_order[k]] < 0) ? 0 : 1;
865  emit_bits_e(entropy, (unsigned int) temp, 1);
866 
867  /* Emit buffered correction bits that must be associated with this code */
868  emit_buffered_bits(entropy, BR_buffer, BR);
869  BR_buffer = entropy->bit_buffer; /* BE bits are gone now */
870  BR = 0;
871  r = 0; /* reset zero run length */
872  }
873 
874  if (r > 0 || BR > 0) { /* If there are trailing zeroes, */
875  entropy->EOBRUN++; /* count an EOB */
876  entropy->BE += BR; /* concat my correction bits to older ones */
877  /* We force out the EOB if we risk either:
878  * 1. overflow of the EOB counter;
879  * 2. overflow of the correction bit buffer during the next MCU.
880  */
881  if (entropy->EOBRUN == 0x7FFF || entropy->BE > (MAX_CORR_BITS-DCTSIZE2+1))
882  emit_eobrun(entropy);
883  }
884 
885  cinfo->dest->next_output_byte = entropy->next_output_byte;
886  cinfo->dest->free_in_buffer = entropy->free_in_buffer;
887 
888  /* Update restart-interval state too */
889  if (cinfo->restart_interval) {
890  if (entropy->restarts_to_go == 0) {
891  entropy->restarts_to_go = cinfo->restart_interval;
892  entropy->next_restart_num++;
893  entropy->next_restart_num &= 7;
894  }
895  entropy->restarts_to_go--;
896  }
897 
898  return TRUE;
899 }

904 LOCAL(boolean)
905 encode_one_block (working_state * state, JCOEFPTR block, int last_dc_val,
906  c_derived_tbl *dctbl, c_derived_tbl *actbl)
907 {
908  register int temp, temp2;
909  register int nbits;
910  register int k, r, i;
911 
912  /* Encode the DC coefficient difference per section F.1.2.1 */
913 
914  temp = temp2 = block[0] - last_dc_val;
915 
916  if (temp < 0) {
917  temp = -temp; /* temp is abs value of input */
918  /* For a negative input, want temp2 = bitwise complement of abs(input) */
919  /* This code assumes we are on a two's complement machine */
920  temp2--;
921  }
922 
923  /* Find the number of bits needed for the magnitude of the coefficient */
924  nbits = 0;
925  while (temp) {
926  nbits++;
927  temp >>= 1;
928  }
929  /* Check for out-of-range coefficient values.
930  * Since we're encoding a difference, the range limit is twice as much.
931  */
932  if (nbits > MAX_COEF_BITS+1)
933  ERREXIT(state->cinfo, JERR_BAD_DCT_COEF);
934 
935  /* Emit the Huffman-coded symbol for the number of bits */
936  if (! emit_bits_s(state, dctbl->ehufco[nbits], dctbl->ehufsi[nbits]))
937  return FALSE;
938 
939  /* Emit that number of bits of the value, if positive, */
940  /* or the complement of its magnitude, if negative. */
941  if (nbits) /* emit_bits rejects calls with size 0 */
942  if (! emit_bits_s(state, (unsigned int) temp2, nbits))
943  return FALSE;
944 
945  /* Encode the AC coefficients per section F.1.2.2 */
946 
947  r = 0; /* r = run length of zeros */
948 
949  for (k = 1; k < DCTSIZE2; k++) {
950  if ((temp = block[jpeg_natural_order[k]]) == 0) {
951  r++;
952  } else {
953  /* if run length > 15, must emit special run-length-16 codes (0xF0) */
954  while (r > 15) {
955  if (! emit_bits_s(state, actbl->ehufco[0xF0], actbl->ehufsi[0xF0]))
956  return FALSE;
957  r -= 16;
958  }
959 
960  temp2 = temp;
961  if (temp < 0) {
962  temp = -temp; /* temp is abs value of input */
963  /* This code assumes we are on a two's complement machine */
964  temp2--;
965  }
966 
967  /* Find the number of bits needed for the magnitude of the coefficient */
968  nbits = 1; /* there must be at least one 1 bit */
969  while ((temp >>= 1))
970  nbits++;
971  /* Check for out-of-range coefficient values */
972  if (nbits > MAX_COEF_BITS)
973  ERREXIT(state->cinfo, JERR_BAD_DCT_COEF);
974 
975  /* Emit Huffman symbol for run length / number of bits */
976  i = (r << 4) + nbits;
977  if (! emit_bits_s(state, actbl->ehufco[i], actbl->ehufsi[i]))
978  return FALSE;
979 
980  /* Emit that number of bits of the value, if positive, */
981  /* or the complement of its magnitude, if negative. */
982  if (! emit_bits_s(state, (unsigned int) temp2, nbits))
983  return FALSE;
984 
985  r = 0;
986  }
987  }
988 
989  /* If the last coef(s) were zero, emit an end-of-block code */
990  if (r > 0)
991  if (! emit_bits_s(state, actbl->ehufco[0], actbl->ehufsi[0]))
992  return FALSE;
993 
994  return TRUE;
995 }

1213 /*
1214  * Generate the best Huffman code table for the given counts, fill htbl.
1215  * Note this is also used by jcphuff.c.
1216  *
1217  * The JPEG standard requires that no symbol be assigned a codeword of all
1218  * one bits (so that padding bits added at the end of a compressed segment
1219  * can't look like a valid code). Because of the canonical ordering of
1220  * codewords, this just means that there must be an unused slot in the
1221  * longest codeword length category. Section K.2 of the JPEG spec suggests
1222  * reserving such a slot by pretending that symbol 256 is a valid symbol
1223  * with count 1. In theory that's not optimal; giving it count zero but
1224  * including it in the symbol set anyway should give a better Huffman code.
1225  * But the theoretically better code actually seems to come out worse in
1226  * practice, because it produces more all-ones bytes (which incur stuffed
1227  * zero bytes in the final file). In any case the difference is tiny.
1228  *
1229  * The JPEG standard requires Huffman codes to be no more than 16 bits long.
1230  * If some symbols have a very small but nonzero probability, the Huffman tree
1231  * must be adjusted to meet the code length restriction. We currently use
1232  * the adjustment method suggested in JPEG section K.2. This method is *not*
1233  * optimal; it may not choose the best possible limited-length code. But
1234  * typically only very-low-frequency symbols will be given less-than-optimal
1235  * lengths, so the code is almost optimal. Experimental comparisons against
1236  * an optimal limited-length-code algorithm indicate that the difference is
1237  * microscopic --- usually less than a hundredth of a percent of total size.
1238  * So the extra complexity of an optimal algorithm doesn't seem worthwhile.
1239  */
1240 
1241 LOCAL(void)
1242 jpeg_gen_optimal_table (j_compress_ptr cinfo, JHUFF_TBL * htbl, long freq[])
1243 {
1244 #define MAX_CLEN 32 /* assumed maximum initial code length */
1245  UINT8 bits[MAX_CLEN+1]; /* bits[k] = # of symbols with code length k */
1246  int codesize[257]; /* codesize[k] = code length of symbol k */
1247  int others[257]; /* next symbol in current branch of tree */
1248  int c1, c2;
1249  int p, i, j;
1250  long v;
1251 
1252  /* This algorithm is explained in section K.2 of the JPEG standard */
1253 
1254  MEMZERO(bits, SIZEOF(bits));
1255  MEMZERO(codesize, SIZEOF(codesize));
1256  for (i = 0; i < 257; i++)
1257  others[i] = -1; /* init links to empty */
1258 
1259  freq[256] = 1; /* make sure 256 has a nonzero count */
1260  /* Including the pseudo-symbol 256 in the Huffman procedure guarantees
1261  * that no real symbol is given code-value of all ones, because 256
1262  * will be placed last in the largest codeword category.
1263  */

1274  v = freq[i];

1277  }

1280  /* In case of ties, take the larger symbol number */
1281  c2 = -1;
1282  v = 1000000000L;

1308  codesize[c2]++;

1311  codesize[c2]++;

1316  for (i = 0; i <= 256; i++) {

1319  /* but I'm paranoid... */
1320  if (codesize[i] > MAX_CLEN)
1321  ERREXIT(cinfo, JERR_HUFF_CLEN_OVERFLOW);

1328  * Huffman procedure assigned any such lengths, we must adjust the coding.
1329  * Here is what the JPEG spec says about how this next bit works:
1330  * Since symbols are paired for the longest Huffman code, the symbols are
1331  * removed from this length category two at a time. The prefix for the pair
1332  * (which is one bit shorter) is allocated to one of the pair; then,
1333  * skipping the BITS entry for that prefix length, a code word from the next
1334  * shortest nonzero BITS entry is converted into a prefix for two code words
1335  * one bit longer.
1336  */

1341  while (bits[j] == 0)

1345  bits[i-1]++; /* one goes in this length */
1346  bits[j+1] += 2; /* two new symbols in this length */
1347  bits[j]--; /* symbol of this length is now a prefix */
1348  }
1349  }

1352  while (bits[i] == 0) /* find largest codelength still in use */
1353  i--;

1357  MEMCOPY(htbl->bits, bits, SIZEOF(htbl->bits));

1361  * changes made above, but the JPEG spec seems to think this works.
1362  */
